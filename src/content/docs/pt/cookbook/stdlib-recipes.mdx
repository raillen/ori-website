Use a referência quando precisar da lista completa de funções.

## Texto e conversão

```ori
import ori.string as strings
import ori.convert.utils as convert_utils

const name: string = strings.title("ori language")
const port: int = convert_utils.parse_int_or("8080", 3000)
```

Use helpers `*_or` quando existe fallback seguro.

## Caminhos e arquivos

```ori
import ori.path as path
import ori.fs.utils as fs_utils

const file_path: string = path.join(["data", "todo.json"])

match fs_utils.exists_result(file_path)
    case success(true):
        io.print("file exists")
    case success(false):
        io.print("file missing")
    case error(message):
        io.print(message)
end
```

## JSON

```ori
import ori.json.utils as json_utils

func read_config(path: string) -> result<json.Value, string>
    const value: json.Value = try json_utils.read(path).or_wrap("reading config")
    return success(value)
end
```

Use `json.utils.read` e `write_pretty` nas bordas de persistência.

Mantenha regras de domínio fora do JSON bruto.

## Processos

```ori
import ori.process as process
import ori.process.utils as process_utils

const capture: map<string, string> = process.run_capture("ori", ["--version"])
const code: int = process_utils.exit_code(capture)
const out: string = process_utils.stdout(capture)
```

Use isso em ferramentas de desenvolvimento, não em lógica de domínio pura.

## Coleções

```ori
import ori.list as lists
import ori.set.utils as set_utils

const total: int = lists.sum_int([1, 2, 3])
const tags: set<string> = set_utils.from_list(["docs", "compiler"])
```

Comece com `list<T>`.

Troque para `queue`, `stack`, `heap`, `tree` ou `graph` quando a estrutura do problema pedir.

## Grafos

```ori
import ori.graph as graph
import ori.graph.algorithms as graph_algorithms

const routes: graph.Graph<string> = graph.new<string>()
graph.add_node(routes, "home")
graph.add_node(routes, "docs")
graph.add_edge(routes, "home", "docs")

const ok: bool = graph_algorithms.has_path(routes, "home", "docs")
```

Use grafos para dependências, rotas e relações entre nós.

## Concorrência

```ori
import ori.task as task
import ori.atomic as atomic

func compute() -> int
    return 41
end

const counter: atomic.AtomicInt = atomic.new(0)
const job: task.Job<int> = task.spawn(do() => compute())

match task.join(job)
    case success(value):
        atomic.add(counter, value)
    case error(_):
        io.print("job failed")
end
```

Use concorrência apenas quando há trabalho independente real.

## Checklist

- Helpers da stdlib ficam perto do problema.
- I/O retorna `result` e é tratado na borda.
- JSON não substitui tipos de domínio.
- Processo externo fica fora de regra pura.
- Estrutura de coleção combina com a operação principal.
